Solve using PyAnsys Math
~~~~~~~~~~~~~~~~~~~~~~~~

    print(f"Solving a ({dim} x {dim}) dense linear system using PyAnsys Math...")

    t1 = time.time()
    s = mm.factorize(a)
    x = s.solve(b, x)
    t2 = time.time()
    pymath_time = t2 - t1
