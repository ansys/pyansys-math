~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~

    def get_res(i):
        """Compute the residual for a given eigenmode."""
        # Eigenfrequency (Hz)
        f = ev[i]

        # omega = 2.pi.Frequency
        omega = 2 * np.pi * f

        # lambda = omega^2
        lam = omega * omega

        # i-th eigenshape
        phi = a[i]

        # K.Phi
        kphi = k.dot(phi)

        # M.Phi
        mphi = m.dot(phi)

        # Normalize scalar value
        kphinrm = kphi.norm()

        # (K-\lambda.M).Phi
        mphi *= lam
        kphi -= mphi

        # Return the residual
        return kphi.norm() / kphinrm


    pymath_acc = np.zeros(nev)

    for i in range(nev):
        f = ev[i]
        pymath_acc[i] = get_res(i)
        print(f"[{i}] : Freq = {f}\t - Residual = {pymath_acc[i]}")
